const int IN1 = 8;
const int IN2 = 9;
const int ENA = 10;   

unsigned long forwardTime = 10000;  
unsigned long backwardTime = 10000; 
unsigned long pauseTime = 2000;     

void setup() {

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENA, OUTPUT);

  analogWrite(ENA, 200);

}

void loop() {

  
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  delay(forwardTime);

 
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  delay(pauseTime);

 
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  delay(backwardTime);

  
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  delay(pauseTime);

}
